import { NextResponse } from "next/server";
import { prisma } from "@/lib/prismaClient";

export async function GET() {
  try {
    const user = await prisma.user.findUnique({
      where: { id: 1 },
      // Aceasta linie exemplifica lazy loading, in care se preiau automat toate cheltuielile rapide asociate unui user
      include: { quick_expenses: true },
      // Putem merge si mai departe, si sa aducem alte informatii despre user, despre membrship-ul lui, despre locatia, etc.
      include: {
        quick_expenses: {
          // alte relatii daca exista
        },
      },
    });

    if (!user) {
      return NextResponse.json({ error: "User not found" }, { status: 404 });
    }

    return NextResponse.json(user);
  } catch (error) {
    console.error("Error fetching user:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
